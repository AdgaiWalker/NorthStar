### 第四阶段：引入 Zustand 状态管理（30分钟）

#### 创建 stores

**`src/stores/useUIStore.ts`** - UI 状态
```typescript
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface UIState {
  themeMode: 'light' | 'eye-care';
  language: 'zh' | 'en' | 'jp' | 'ru';
  currentDomain: 'creative' | 'dev' | 'work';
  
  setThemeMode: (mode: 'light' | 'eye-care') => void;
  toggleTheme: () => void;
  setLanguage: (lang: 'zh' | 'en' | 'jp' | 'ru') => void;
  setDomain: (domain: 'creative' | 'dev' | 'work') => void;
}

export const useUIStore = create<UIState>()(
  persist(
    (set) => ({
      themeMode: 'light',
      language: 'zh',
      currentDomain: 'creative',
      
      setThemeMode: (mode) => set({ themeMode: mode }),
      toggleTheme: () => set((state) => ({ 
        themeMode: state.themeMode === 'light' ? 'eye-care' : 'light' 
      })),
      setLanguage: (lang) => set({ language: lang }),
      setDomain: (domain) => set({ currentDomain: domain }),
    }),
    { name: 'pangen-ui-storage' }
  )
);
```

**`src/stores/useAuthStore.ts`** - 认证状态
```typescript
import { create } from 'zustand';

interface AuthState {
  isLoggedIn: boolean;
  user: null | { id: string; name: string; email: string };
  
  login: (user: { id: string; name: string; email: string }) => void;
  logout: () => void;
}

export const useAuthStore = create<AuthState>((set) => ({
  isLoggedIn: false,
  user: null,
  
  login: (user) => set({ isLoggedIn: true, user }),
  logout: () => set({ isLoggedIn: false, user: null }),
}));
```

**`src/stores/useSolutionStore.ts`** - 方案状态
```typescript
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface SolutionState {
  selectedToolIds: Set<string>;
  userSolutions: UserSolution[];
  
  toggleToolSelection: (toolId: string) => void;
  clearSelection: () => void;
  addSolution: (solution: UserSolution) => void;
  deleteSolution: (id: string) => void;
}

export const useSolutionStore = create<SolutionState>()(
  persist(
    (set) => ({
      selectedToolIds: new Set<string>(),
      userSolutions: [],
      
      toggleToolSelection: (toolId) => set((state) => {
        const newSet = new Set(state.selectedToolIds);
        if (newSet.has(toolId)) {
          newSet.delete(toolId);
        } else {
          newSet.add(toolId);
        }
        return { selectedToolIds: newSet };
      }),
      
      clearSelection: () => set({ selectedToolIds: new Set() }),
      
      addSolution: (solution) => set((state) => ({
        userSolutions: [solution, ...state.userSolutions]
      })),
      
      deleteSolution: (id) => set((state) => ({
        userSolutions: state.userSolutions.filter(s => s.id !== id)
      })),
    }),
    { 
      name: 'pangen-solution-storage',
      // 处理 Set 序列化
      partialize: (state) => ({
        selectedToolIds: Array.from(state.selectedToolIds),
        userSolutions: state.userSolutions,
      }),
      onRehydrateStorage: () => (state) => {
        if (state) {
          state.selectedToolIds = new Set(state.selectedToolIds as any);
        }
      },
    }
  )
);
```

import { BrowserRouter, Routes, Route } from 'react-router-dom';
import { Header } from './components/layout/Header';
import { Footer } from './components/layout/Footer';

// Pages
import HomePage from './pages/HomePage';
import ToolDetailPage from './pages/ToolDetailPage';
import ArticleReadPage from './pages/ArticleReadPage';
import LoginPage from './pages/LoginPage';
import UserCenterPage from './pages/UserCenterPage';
import SolutionGeneratePage from './pages/SolutionGeneratePage';
import AdminPage from './pages/AdminPage';

function App() {
  return (
    <BrowserRouter>
      <div className="min-h-screen flex flex-col">
        <Header />
        <main className="flex-grow pt-6">
          <Routes>
            <Route path="/" element={<HomePage />} />
            <Route path="/tool/:toolId" element={<ToolDetailPage />} />
            <Route path="/article/:articleId" element={<ArticleReadPage />} />
            <Route path="/login" element={<LoginPage />} />
            <Route path="/user-center/*" element={<UserCenterPage />} />
            <Route path="/solution/generate" element={<SolutionGeneratePage />} />
            <Route path="/admin/*" element={<AdminPage />} />
          </Routes>
        </main>
        <Footer />
      </div>
    </BrowserRouter>
  );
}

export default App;
